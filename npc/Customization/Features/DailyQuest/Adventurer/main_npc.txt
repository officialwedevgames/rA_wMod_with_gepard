	function	quest_npc_loc	{

			set .@num, quest_relay;
			
			deletearray .@TT; deletearray .@XT$; 
			
			explode(.@XT$,$npclocation$[.@num],"|");

			for(set .@l,0; .@l < getarraysize(.@XT$); set .@l,.@l + 1)
			{
				set .@TT[.@l], atoi(.@XT$[.@l]);
			}
			
			set .@location$,""+.@XT$[0]+" "+.@XT$[1]+" ("+.@TT[2]+", "+.@TT[3]+")";

		return .@location$;
	}
